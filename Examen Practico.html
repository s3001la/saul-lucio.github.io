Examen Práctico: Registro de notas y asistencias

📌 Aplicación con interfaz gráfica (Tkinter) que permite:

  • Agregar estudiantes a una lista
  • Registrar notas (múltiples por estudiante)
  • Registrar asistencia (contador)
  • Mostrar promedio y datos completos
# Examen Práctico - Proyecto de Aplicación # Registro de notas y asistencias estudiantiles con Tkinter import tkinter as tk # Diccionario para guardar estudiantes estudiantes = {} # Función para agregar estudiante def agregar_estudiante(): nombre = entry_nombre.get() if nombre != "": estudiantes[nombre] = {"notas": [], "asistencia": 0} lista.insert(tk.END, nombre) entry_nombre.delete(0, tk.END) # Función para agregar nota def agregar_nota(): estudiante = lista.get(tk.ACTIVE) nota = entry_nota.get() if estudiante != "" and nota != "": estudiantes[estudiante]["notas"].append(float(nota)) entry_nota.delete(0, tk.END) # Función para registrar asistencia def registrar_asistencia(): estudiante = lista.get(tk.ACTIVE) if estudiante != "": estudiantes[estudiante]["asistencia"] += 1 # Función para mostrar resultados def mostrar_datos(): estudiante = lista.get(tk.ACTIVE) if estudiante != "": notas = estudiantes[estudiante]["notas"] asistencia = estudiantes[estudiante]["asistencia"] if len(notas) > 0: promedio = sum(notas) / len(notas) else: promedio = 0 resultado.config(text=f"Estudiante: {estudiante}\nNotas: {notas}\nPromedio: {promedio:.2f}\nAsistencia: {asistencia}") # Ventana principal ventana = tk.Tk() ventana.geometry("350x450") ventana.title("Registro Estudiantil") # Frame contenedor frame = tk.Frame(ventana, bg="grey") frame.pack(fill="both", expand=True) # === INGRESO DE NOMBRE === tk.Label(frame, text="Nombre del estudiante", bg="black", fg="white").pack() entry_nombre = tk.Entry(frame) entry_nombre.pack() tk.Button(frame, text="Agregar estudiante", bg="green", fg="white", command=agregar_estudiante).pack() # === LISTA DE ESTUDIANTES === lista = tk.Listbox(frame) lista.pack() # === INGRESO DE NOTA === tk.Label(frame, text="Ingrese nota", bg="black", fg="white").pack() entry_nota = tk.Entry(frame) entry_nota.pack() tk.Button(frame, text="Agregar nota", bg="blue", fg="white", command=agregar_nota).pack() # === ASISTENCIA === tk.Button(frame, text="Registrar asistencia", bg="orange", fg="white", command=registrar_asistencia).pack() # === MOSTRAR RESULTADOS === tk.Button(frame, text="Mostrar datos", bg="purple", fg="white", command=mostrar_datos).pack() # === ÁREA DE RESULTADO === resultado = tk.Label(frame, text="Resultado", bg="green", fg="white") resultado.pack() ventana.mainloop()
