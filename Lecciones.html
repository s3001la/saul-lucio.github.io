Calculadora IMC con Tkinter

📌 Interfaz gráfica para calcular el Índice de Masa Corporal y clasificarlo según la OMS.

import tkinter as tk def calcular_imc(): peso = float(entry_peso.get()) altura = float(entry_altura.get()) imc = peso / (altura ** 2) if imc < 18.5: categoria = "Bajo peso" elif imc < 25: categoria = "Normal" elif imc < 30: categoria = "Sobrepeso" else: categoria = "Obesidad" resultado.config(text=f"IMC: {imc:.2f} - {categoria}") ventana = tk.Tk() ventana.geometry("300x300") ventana.title("Calculadora IMC") frame = tk.Frame(ventana, bg="grey") frame.pack(fill="both", expand=True) label_peso = tk.Label(frame, text="Ingrese su peso (kg)", bg="black", fg="white", font=("Cambria", 12)) label_peso.pack() entry_peso = tk.Entry(frame) entry_peso.pack() label_altura = tk.Label(frame, text="Ingrese su altura (m)", bg="black", fg="white", font=("Cambria", 12)) label_altura.pack() entry_altura = tk.Entry(frame) entry_altura.pack() button = tk.Button(frame, text="Calcular IMC", bg="green", fg="white", font=("Arial", 12), command=calcular_imc) button.pack() resultado = tk.Label(frame, text="Resultado", bg="green", fg="white", font=("Arial", 12)) resultado.pack() ventana.mainloop()

Calculadora Básica con Tkinter

📌 Interfaz gráfica para suma, resta, multiplicación y división usando botones.

import tkinter as tk def obtener_valores(): num1 = float(entry1.get()) num2 = float(entry2.get()) return num1, num2 def suma(): num1, num2 = obtener_valores() resultado.config(text=f"Resultado: {num1 + num2}") def resta(): num1, num2 = obtener_valores() resultado.config(text=f"Resultado: {num1 - num2}") def multiplicacion(): num1, num2 = obtener_valores() resultado.config(text=f"Resultado: {num1 * num2}") def division(): num1, num2 = obtener_valores() if num2 == 0: resultado.config(text="Error: División entre cero") else: resultado.config(text=f"Resultado: {round(num1 / num2, 2)}") ventana = tk.Tk() ventana.geometry("300x350") ventana.title("Calculadora Básica") frame = tk.Frame(ventana, bg="grey") frame.pack(fill="both", expand=True) label1 = tk.Label(frame, text="Ingrese el 1er número", bg="black", fg="white", font=("Cambria", 12)) label1.pack() entry1 = tk.Entry(frame) entry1.pack() label2 = tk.Label(frame, text="Ingrese el 2do número", bg="black", fg="white", font=("Cambria", 12)) label2.pack() entry2 = tk.Entry(frame) entry2.pack() tk.Button(frame, text="Sumar", bg="green", fg="white", font=("Gothic", 12), command=suma).pack() tk.Button(frame, text="Restar", bg="green", fg="white", font=("Gothic", 12), command=resta).pack() tk.Button(frame, text="Multiplicar", bg="green", fg="white", font=("Gothic", 12), command=multiplicacion).pack() tk.Button(frame, text="Dividir", bg="green", fg="white", font=("Gothic", 12), command=division).pack() resultado = tk.Label(frame, text="Resultado: ", bg="red", fg="white", font=("Arial", 12)) resultado.pack() ventana.mainloop()

Calculadora Básica en Java (consola)

📌 Uso de switch, do-while y funciones para operaciones aritméticas.

import java.util.Scanner; public class Calculadora { public static double suma(double a, double b) { return a + b; } public static double resta(double a, double b) { return a - b; } public static double multiplicacion(double a, double b) { return a * b; } public static double division(double a, double b) { return b != 0 ? a / b : 0; } public static void main(String[] args) { Scanner sc = new Scanner(System.in); int opcion; do { System.out.println("\n--- CALCULADORA ---"); System.out.println("1. Suma"); System.out.println("2. Resta"); System.out.println("3. Multiplicación"); System.out.println("4. División"); System.out.println("5. Salir"); System.out.print("Elige una opción: "); opcion = sc.nextInt(); if (opcion >= 1 && opcion <= 4) { System.out.print("Ingrese el primer número: "); double num1 = sc.nextDouble(); System.out.print("Ingrese el segundo número: "); double num2 = sc.nextDouble(); double resultado = 0; switch (opcion) { case 1: resultado = suma(num1, num2); break; case 2: resultado = resta(num1, num2); break; case 3: resultado = multiplicacion(num1, num2); break; case 4: if (num2 == 0) System.out.println("Error: No se puede dividir entre 0"); else resultado = division(num1, num2); break; } System.out.println("Resultado: " + resultado); } } while (opcion != 5); System.out.println("Programa finalizado."); sc.close(); } }
